// Verify nonce request result via Google API
export const verifySignatureWithAPI: VerifySignature = async (
  signedAttestation,
  apiKey
) => {
  if (!apiKey) {
    return PASS_ON_FAILURE;
  }
  const API_URL = `https://www.googleapis.com/androidcheck/v1/attestations/verify?key=${apiKey}`;
 
  try {
    const verificationResult = await fetch(API_URL, {
      method: 'POST',
      compress: false,
      body: JSON.stringify({ signedAttestation }),
    }).then(response => {
      switch (response.status) {
        case 200:
          return response.json();
        case 429: // API rate limit reached
          return { isValidSignature: PASS_ON_FAILURE };
        default:
          return { isValidSignature: PASS_ON_FAILURE };
      }
    });
 
    return verificationResult.isValidSignature;
  } catch (err) {
    return PASS_ON_FAILURE;
  }
};
 
// Verify Jws with Google Safety API (Online verify) Subject to API rate limits
export const verifyDeviceWithAPI: VerifyDevice = async (
  generatedNonce,
  signedAttestation,
  apiKey
) => {
  const isValidSignature = await verifySignatureWithAPI(
    signedAttestation,
    apiKey
  );
 
  if (!isValidSignature) {
    return false;
  }
 
  const decodedJws = jws.decode(signedAttestation);
  const payload = JSON.parse(decodedJws.payload);
  const result = generatedNonce === payload.nonce && payload.basicIntegrity;
 
  return result;
};
